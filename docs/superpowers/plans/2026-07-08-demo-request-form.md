- [ ] **Step 5: Implement validation helpers**

Create `src/lib/demoRequest.ts`:

```ts
export type DemoRequestValues = {
  name: string;
  email: string;
  storeUrl: string;
  country: string;
  storeSize: string;
  message: string;
};

export type DemoRequestErrors = Partial<Record<keyof DemoRequestValues, string>>;

export type DemoRequestPayload = DemoRequestValues & {
  submittedAt: string;
  source: "kukis-site";
};

const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

function normalizeUrl(value: string) {
  const trimmed = value.trim();
  if (!trimmed) return "";
  return /^https?:\/\//i.test(trimmed) ? trimmed : `https://${trimmed}`;
}

function isValidStoreUrl(value: string) {
  const normalized = normalizeUrl(value);
  try {
    const url = new URL(normalized);
    return Boolean(url.hostname.includes("."));
  } catch {
    return false;
  }
}

export function validateDemoRequest(values: DemoRequestValues): DemoRequestErrors {
  const errors: DemoRequestErrors = {};

  if (!values.name.trim()) errors.name = "Enter your name.";
  if (!values.email.trim()) {
    errors.email = "Enter a work email.";
  } else if (!emailPattern.test(values.email.trim())) {
    errors.email = "Enter a valid email address.";
  }
  if (!values.storeUrl.trim()) {
    errors.storeUrl = "Enter your Shopify store URL.";
  } else if (!isValidStoreUrl(values.storeUrl)) {
    errors.storeUrl = "Enter a valid store URL or domain.";
  }
  if (!values.country.trim()) errors.country = "Choose a country.";
  if (!values.message.trim()) errors.message = "Tell us what we should look at.";

  return errors;
}

export function hasDemoRequestErrors(errors: DemoRequestErrors) {
  return Object.keys(errors).length > 0;
}

export function buildDemoRequestPayload(values: DemoRequestValues): DemoRequestPayload {
  return {
    name: values.name.trim(),
    email: values.email.trim(),
    storeUrl: normalizeUrl(values.storeUrl),
    country: values.country.trim(),
    storeSize: values.storeSize.trim(),
    message: values.message.trim(),
    submittedAt: new Date().toISOString(),
    source: "kukis-site",
  };
}

export function buildDemoRequestMailto(email: string, payload: DemoRequestPayload) {
  const subject = encodeURIComponent(`Kukis demo request from ${payload.name}`);
  const body = encodeURIComponent(
    [
      `Name: ${payload.name}`,
      `Email: ${payload.email}`,
      `Store URL: ${payload.storeUrl}`,
      `Country: ${payload.country}`,
      `Store size: ${payload.storeSize || "Not provided"}`,
      "",
      "Message:",
      payload.message,
      "",
      `Submitted at: ${payload.submittedAt}`,
    ].join("\n")
  );

  return `mailto:${email}?subject=${subject}&body=${body}`;
}
```

Create `src/components/ui/CookieMascot.tsx`:

```ts
import { useEffect, useRef } from "react";
import { motion, useMotionValue, useSpring } from "framer-motion";
import { cn } from "../../lib/cn";
import { publicAsset } from "../../lib/publicAsset";
import { usePrefersReducedMotion } from "../../lib/useReducedMotion";

export type CookieMascotMood = "idle" | "promising" | "happy";

export function CookieMascot({
  bubble,
  mood = "idle",
  className,
}: {
  bubble: string;
  mood?: CookieMascotMood;
  className?: string;
}) {
  const reduced = usePrefersReducedMotion();
  const ref = useRef<HTMLDivElement | null>(null);
  const px = useMotionValue(0);
  const py = useMotionValue(0);
  const spring = { stiffness: 220, damping: 18, mass: 0.4 };
  const sx = useSpring(px, spring);
  const sy = useSpring(py, spring);
  const happy = mood === "happy";

  useEffect(() => {
    if (reduced) return;
    const onMove = (e: MouseEvent) => {
      const el = ref.current;
      if (!el) return;
      const r = el.getBoundingClientRect();
      const cx = r.left + r.width * 0.5;
      const cy = r.top + r.height * 0.48;
      const dx = e.clientX - cx;
      const dy = e.clientY - cy;
      const dist = Math.hypot(dx, dy) || 1;
      const max = r.width * 0.032;
      px.set((dx / dist) * max);
      py.set((dy / dist) * max);
    };
    window.addEventListener("mousemove", onMove);
    return () => window.removeEventListener("mousemove", onMove);
  }, [reduced, px, py]);

  const pupilCls = "pointer-events-none absolute rounded-full bg-[#2a170c]";
  const pupilStyle = { width: "8.5%", height: "8.5%", x: sx, y: sy } as const;

  return (
    <div className={cn("relative w-[132px] shrink-0 md:w-[200px]", className)}>
      <motion.div
        key={bubble}
        initial={reduced ? false : { opacity: 0, y: 6, scale: 0.92 }}
        animate={{ opacity: 1, y: 0, scale: 1 }}
        transition={{ duration: 0.3, ease: [0.22, 1, 0.36, 1] }}
        className="absolute -top-2 left-1/2 z-10 max-w-[180px] -translate-x-1/2 -translate-y-full rounded-2xl border border-chip/15 bg-milk px-3 py-1.5 text-center text-[12px] font-semibold leading-tight text-cocoa shadow-soft"
      >
        {bubble}
        <span className="absolute -bottom-[5px] left-1/2 h-2.5 w-2.5 -translate-x-1/2 rotate-45 border-b border-r border-chip/15 bg-milk" />
      </motion.div>

      <motion.div
        ref={ref}
        className="relative"
        animate={reduced ? { y: 0, rotate: 0 } : happy ? "happy" : "idle"}
        variants={{
          idle: { y: [0, -5, 0], rotate: 0 },
          happy: { y: [0, -16, 0], rotate: [0, -7, 7, 0] },
        }}
        transition={
          happy
            ? { duration: 0.6, ease: "easeOut" }
            : { duration: 3, repeat: reduced ? 0 : Infinity, ease: "easeInOut" }
        }
      >
        <img
          src={publicAsset("mascot/cookie.png")}
          alt="Kukis cookie mascot"
          className="w-full select-none drop-shadow-[0_14px_22px_rgba(90,52,32,0.22)]"
          draggable={false}
        />
        <motion.div className={pupilCls} style={{ ...pupilStyle, left: "31.4%", top: "44.5%" }} />
        <motion.div className={pupilCls} style={{ ...pupilStyle, left: "58.6%", top: "43.5%" }} />
      </motion.div>
    </div>
  );
}
```
